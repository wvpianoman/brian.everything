## after a successful installation, you need to run

# Update the system

sudo xbps-install -Su

## Install OctoXBPS, void linux package manager  Similar to synaptic for Debian.

sudo xbps-install -S octoxbps

#OK, lets install some software

sudo xbps-install -S cowsay yad duf espeak-ng figlet fortune-mod fortune-mod-anarchism fortune-mod-void pandoc aria2 thefuck pdfsandwich autoconf automake bc binutils btop busybox cifs-utils codec2 dconf-editor direnv earlyoom easyeffects mbpfan fd ffmpeg ffmpegthumbnailer ffmpegthumbs flatpak git gnupg2 grep haveged ibus-gtk4 jq lsd make meld mpg123 nano net-tools nftables p7zip pandoc bat tldr tesseract-ocr-eng tesseract-ocr tesseract-ocr-eng

sudo xbps-install -S fish-shell bash-completion checkbashisms powerline-go zsh zsh-autosuggestions zsh-completions zsh-syntax-highlighting zsh-history-substring-search

# using this to make sure that all of cinnamon is installed

sudo xbps-install -S cinnamon-all

# add extra repos to void and install some packages

echo "repository=https://github.com/noid-linux/xbps-repo/releases/latest/download" | sudo tee /etc/xbps.d/noid-xbps-repo.conf
sudo xbps-install -S void-repo-nonfree
sudo xbps-install -Su
